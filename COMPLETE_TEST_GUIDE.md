**When:** Step 1 - When you POST to `/tasks`

**File:** `/home/mrueda/WWW/interview/python/app/main.py`

**Code Location:** Lines 39-75

```python
@app.post("/tasks", response_model=TaskResponse, status_code=status.HTTP_201_CREATED)
async def create_task(
    task: TaskCreate,
    db: Session = Depends(get_db),
    sqs_client: SQSClient = Depends(get_sqs_client)
):
    """Create a new task and send it to SQS for processing"""
    
    # ‚úÖ STEP 1: Create task in database
    db_task = Task(
        title=task.title,
        description=task.description,
        status=TaskStatus.PENDING  # ‚Üê Status starts as "pending"
    )
    db.add(db_task)
    db.commit()
    db.refresh(db_task)
    
    # ‚úÖ STEP 2: Send message to SQS
    try:
        message = {
            "task_id": db_task.id,      # ‚Üê Task ID
            "title": db_task.title,
            "description": db_task.description
        }
        sqs_client.send_message(message)  # ‚Üê Sends to SQS
    except Exception as e:
        # If SQS fails, mark task as failed
        db_task.status = TaskStatus.FAILED
        db_task.result = f"Failed to send to SQS: {str(e)}"
        db.commit()
        raise HTTPException(status_code=500, detail=str(e))
    
    # ‚úÖ STEP 3: Return response to client
    return db_task
```

**What happens here:**
1. ‚úÖ Validates input (Pydantic)
2. ‚úÖ Creates task in PostgreSQL (status: "pending")
3. ‚úÖ Sends message to SQS queue
4. ‚úÖ Returns response immediately (doesn't wait for processing)

def send_message(self, message_body: dict):
    """Send a message to SQS queue"""
    try:
        # ‚úÖ Convert Python dict to JSON string
        response = self.client.send_message(
            QueueUrl=self.queue_url,  # ‚Üê http://localstack:4566/.../interview-queue
            MessageBody=json.dumps(message_body)  # ‚Üê {"task_id": 6, "title": "..."}
        )
        return response
    except Exception as e:
        print(f"Error sending message to SQS: {e}")
        raise
```

**What happens here:**
1. ‚úÖ Converts message to JSON string
2. ‚úÖ Uses boto3 to send to SQS
3. ‚úÖ SQS stores the message
4. ‚úÖ Returns MessageId

def lambda_handler(event, context):
    """
    AWS Lambda handler function
    Processes messages from SQS queue and updates task status in PostgreSQL
    """
    print(f"Received event: {json.dumps(event)}")
    
    processed_records = []
    failed_records = []
    
    # ‚úÖ STEP 1: Loop through SQS messages
    for record in event['Records']:
        try:
            # ‚úÖ STEP 2: Parse SQS message
            message_body = json.loads(record['body'])
            task_id = message_body.get('task_id')      # ‚Üê 6
            title = message_body.get('title')          # ‚Üê "Test Lambda Flow"
            description = message_body.get('description')
            
            print(f"Processing task {task_id}: {title}")
            
            # ‚úÖ STEP 3: Connect to database
            conn = get_db_connection()
            cursor = conn.cursor()
            
            # ‚úÖ STEP 4: Update task status to PROCESSING
            cursor.execute(
                "UPDATE tasks SET status = %s, updated_at = %s WHERE id = %s",
                ('processing', datetime.now(), task_id)
            )
            conn.commit()
            
            # ‚úÖ STEP 5: Process the task (business logic)
            result = process_task(task_id, title, description)
            # result = "Processed task 'Test Lambda Flow' at 2025-10-23..."
            
            # ‚úÖ STEP 6: Update task status to COMPLETED
            cursor.execute(
                """
                UPDATE tasks 
                SET status = %s, result = %s, updated_at = %s 
                WHERE id = %s
                """,
                ('completed', result, datetime.now(), task_id)
            )
            conn.commit()
            
            # ‚úÖ STEP 7: Cleanup
            cursor.close()
            conn.close()
            
            processed_records.append(task_id)
            print(f"Successfully processed task {task_id}")
            
        except Exception as e:
            # Handle errors
            error_msg = str(e)
            print(f"Error processing record: {error_msg}")
            failed_records.append({
                'task_id': task_id if 'task_id' in locals() else 'unknown',
                'error': error_msg
            })
            
            # Try to update task status to FAILED
            try:
                conn = get_db_connection()
                cursor = conn.cursor()
                cursor.execute(
                    """
                    UPDATE tasks 
                    SET status = %s, result = %s, updated_at = %s 
                    WHERE id = %s
                    """,
                    ('failed', f"Error: {error_msg}", datetime.now(), task_id)
                )
                conn.commit()
                cursor.close()
                conn.close()
            except Exception as db_error:
                print(f"Failed to update task status: {db_error}")
    
    # ‚úÖ STEP 8: Return success response
    response = {
        'statusCode': 200,
        'body': json.dumps({
            'message': 'Processing complete',
            'processed': processed_records,
            'failed': failed_records
        })
    }
    
    return response
```

**What happens here:**
1. ‚úÖ Receives event from SQS (automatic trigger)
2. ‚úÖ Parses message body
3. ‚úÖ Connects to PostgreSQL database
4. ‚úÖ Updates status to "processing"
5. ‚úÖ Executes business logic
6. ‚úÖ Updates status to "completed" with result
7. ‚úÖ Returns success (SQS auto-deletes message)
